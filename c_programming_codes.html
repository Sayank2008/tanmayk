Program 2

Description: This program checks whether a number is even or odd.

  else

}

printf("\nThe circumference of the circle is %f", circumference);

Program 7

Description: This program swaps two numbers

#include
int main ()
{
int a,b,temp;
printf("\nEnter 1st number: ");
scanf("%d",&a);
printf("\nEnter 2nd number: ");
scanf("%d",&b);
printf("\nThe numbers before swap are %d and %d",a,b);
temp=a;
a=b;
b=temp;
printf("\nThe numbers after swap are %d and %d",a,b);
return 0;
}
