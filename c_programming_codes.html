Description:This program calculates the sum of two numbers.

Program 2

Description: This program checks whether a number is even or odd.

#include<stdio.h>
int main() {
  int num;
  printf("Enter a number: ");
  scanf("%d", &num);
  if(num % 2 == 0)
    printf("Even");
  else
    printf("Odd");
  return 0;
}
